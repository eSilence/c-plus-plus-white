
class Function_part{
public:
	Function_part(char new_opration, double new_value){
		operation = new_opration;
		value = new_value;
	}
	double Apply(double new_value) const {
		if (operation == '+')
			return value + new_value;
		else if (operation == '-')
			return new_value - value;
		else if (operation == '*')
			return new_value * value;
		return new_value / value;
	}
	void Invert(){
		if (operation == '+')
			operation = '-';
		else if (operation == '-')
			operation = '+';
		else if (operation == '*')
			operation = '/';
		else
			operation = '*';
	}
private:
	char operation;
	double value;
};

class Function{
public:
	void AddPart(char new_opration, double new_value){
		part.push_back( Function_part(new_opration, new_value));
	}
	double Apply(double value) const {
		for (const Function_part &p: part)
			value = p.Apply(value);
		return value;
	}
	void Invert(){
		for (Function_part &p: part)
			p.Invert();
		reverse(part.begin(), part.end());
	}
private:
	vector<Function_part> part;

};
